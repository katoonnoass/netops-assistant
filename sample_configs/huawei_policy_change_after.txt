#
sysname NE40-POLICY-DIFF-AFTER
#
interface GigabitEthernet0/0/0
 description UPLINK-CORE
 ip address 10.255.0.1 255.255.255.252
#
interface GigabitEthernet0/0/1
 description CLIENTE-X-V2
 ip address 10.0.1.1 255.255.255.252
#
interface GigabitEthernet0/0/2
 description NOVO-CLIENTE-Y
 ip address 10.0.2.1 255.255.255.252
#
interface LoopBack0
 ip address 192.168.0.1 255.255.255.255
#
ip ip-prefix CLIENTE-X index 10 permit 200.200.200.0 30
ip ip-prefix CLIENTE-X index 20 permit 200.200.201.0 28
ip ip-prefix NOVO-PREFIXO index 10 permit 10.0.0.0 8
#
ip as-path-filter 10 permit ^64520$
ip as-path-filter 10 permit ^64600$
ip as-path-filter 10 deny .*
#
ip community-filter 20 permit 65000:100
ip community-filter 20 permit 65000:200
#
route-policy EXPORT-CLIENTE permit node 10
 if-match ip-prefix CLIENTE-X
 apply community 65000:100 65000:200 additive
#
route-policy EXPORT-CLIENTE deny node 20
 if-match ip-prefix NOVO-PREFIXO
#
route-policy IMPORT-CLIENTE permit node 10
 if-match as-path-filter 10
 if-match community-filter 20
 apply local-preference 200
#
route-policy NOVA-POLICY permit node 10
 apply local-preference 150
#
acl number 3001
 rule 5 permit ip source 10.0.0.0 0.0.0.255 destination 192.168.0.0 0.0.255.255
 rule 10 deny ip
#
bgp 65000
 bgp router-id 192.168.0.1
 peer 10.255.0.2 as-number 64500
 peer 10.255.0.2 description UPSTREAM
 peer 10.255.0.2 route-policy EXPORT-CLIENTE export
 peer 10.255.0.2 route-policy IMPORT-CLIENTE import
 peer 10.0.2.2 as-number 64530
 peer 10.0.2.2 description CLIENTE-Y
 peer 10.0.2.2 route-policy NOVA-POLICY export
#
return
